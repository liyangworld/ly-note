判断一个对象是否为空 {} 的三种方法

1. for ... in 

function isEmptyObject(obj){
    for(var key in obj){ return true }
    return false;
}

2. JSON.stringify()

function isEmptyObject(obj){
    if(JSON.stringify(obj) === '{}'){ return true }
    return false;
}

3. ES5的Object.keys()

function isEmptyObject(obj){
    if(Object.keys(obj).length === 0){ return true }
    return false;
}

封装 AJAX 请求

1. 参数连接

function _joinParams(obj,isSkipEmpty) {
    var str = [];
    for (var key in obj) {
        if (isSkipEmpty) {
            if (obj[key] === null || obj[key] === '') continue;
        }
        str.push(key + "=" + obj[key]);
    }
    return str.join("&");
};

2. get 请求

var get = function(url, params,successFn,errorFn) {
        var serverURL = url;
        var paramsStr = _joinParams(params,true);

        if (paramsStr) {
            if (serverURL.indexOf('?') > 0) {
                serverURL += '&' + paramsStr;
            }else{
                serverURL += '?' + paramsStr;
            }
        }
        $.ajax({
            url: serverURL,
            method: 'GET',
            dataType: 'json',
            data: null,
            cache: false,
            crossDomain: true,
            complete: function (jqXHR, textStatus) {
            },
            success: function (responseData, textStatus, jqXHR) {
                successFn && successFn(responseData);
            },
            error: function (jqXHR, textStatus, errorThrown) {
                errorFn && errorFn();
            }
        });
    }

var post = function (url, data, successFn, errorFn) {
        var postData = JSON.stringify(data);
        $.ajax({
            url: url,
            method: 'POST',
            dataType: 'json',
            contentType: "application/json",
