SQL命令速览手册

简介

SQL 命令对大小写不敏感，即大小写都可以。

分号是在数据库系统中分隔每条 SQL 语句的标准方法，并且某些数据库系统要求在每条 SQL 语句的末端都使用分号，因此建议始终使用分号。

速览手册

SQL 命令 用法
show variables like '%char%'; 查看数据库所设置的字符集变量，wins下配置文件为my.ini
set names utf8; 设置某数据库的客户端传来和接受的数据所使用的字符集
show databases show databases
show tables show tables
use database_name use database_name
create database CREATE DATABASE database_name
create table CREATE TABLE table_name ( column_name1 data_type, column_name2 data_type, ... )
create view CREATE VIEW view_name AS SELECT column_name(s) FROM table_name WHERE condition
create index CREATE INDEX index_name ON table_name (column_name) 或 CREATE UNIQUE INDEX index_name ON table_name (column_name)
drop database DROP DATABASE database_name
drop table DROP TABLE table_name
truncate table TRUNCATE TABLE table_name。仅仅删除表内的数据，但并不删除表本身，索引将重置为初始值
drop index DROP INDEX table_name.index_name (SQL Server) 或 ALTER TABLE table_name DROP INDEX index_name (MySQL)
delete DELETE FROM table_name WHERE some_column=some_value 或 DELETE FROM table_name (Note: Deletes the entire table!!) 或 DELETE * FROM table_name (Note: Deletes the entire table!!)
alter table ALTER TABLE table_name ADD column_name datatype 或 ALTER TABLE table_name DROP COLUMN column_name 或 ALTER TABLE table_name MODIFY COLUMN column_name datatype（MySQL） 或 ALTER TABLE table_name ALTER COLUMN column_name datatype（SQL Server）
insert into INSERT INTO table_name VALUES (value1, value2, value3,....) 或 INSERT INTO table_name (column1, column2, column3,...) VALUES (value1, value2, value3,....)
update UPDATE table_name SET column1=value, column2=value,... WHERE some_column=some_value
select SELECT column_name(s) FROM table_name
select * SELECT * FROM table_name
select distinct SELECT DISTINCT column_name(s) FROM table_name
select top SELECT TOP number|percent column_name(s) FROM table_name
select into SELECT * INTO new_table_name [IN externaldatabase] FROM old_table_name 或 SELECT column_name(s) INTO new_table_name [IN externaldatabase] FROM old_table_name
as (alias) SELECT column_name AS column_alias FROM table_name 或 SELECT column_name FROM table_name AS table_alias
where SELECT column_name(s) FROM table_name WHERE column_name operator value
and、or SELECT column_name(s) FROM table_name WHERE condition AND|OR condition
between SELECT column_name(s) FROM table_name WHERE column_name BETWEEN value1 AND value2
in SELECT column_name(s) FROM table_name WHERE column_name IN (value1,value2,..)
like SELECT column_name(s) FROM table_name WHERE column_name LIKE pattern
group by SELECT column_name, aggregate_function(column_name) FROM table_name WHERE column_name operator value GROUP BY column_name
having SELECT column_name, aggregate_function(column_name) FROM table_name WHERE column_name operator value GROUP BY column_name HAVING aggregate_function(column_name) operator value
order by SELECT column_name(s) FROM table_name ORDER BY column_name [ASC|DESC]
inner join SELECT column_name(s) FROM table_name1 INNER JOIN table_name2 ON table_name1.column_name=table_name2.column_name
left join SELECT column_name(s) FROM table_name1 LEFT JOIN table_name2 ON table_name1.column_name=table_name2.column_name
right join SELECT column_name(s) FROM table_name1 RIGHT JOIN table_name2 ON table_name1.column_name=table_name2.column_name
full join SELECT column_name(s) FROM table_name1 FULL JOIN table_name2 ON table_name1.column_name=table_name2.column_name
union SELECT column_name(s) FROM table_name1 UNION SELECT column_name(s) FROM table_name2
union all SELECT column_name(s) FROM table_name1 UNION ALL SELECT column_name(s) FROM table_name2
